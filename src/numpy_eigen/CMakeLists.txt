cmake_minimum_required(VERSION 3.5)
project(numpy_eigen)

# Default to C++17
if(NOT CMAKE_CXX_STANDARD)
  set(CMAKE_CXX_STANDARD 17)
endif()

if(CMAKE_COMPILER_IS_GNUCXX OR CMAKE_CXX_COMPILER_ID MATCHES "Clang")
  add_compile_options(-Wall -Wextra -Wpedantic)
endif()

# Find dependencies
find_package(ament_cmake REQUIRED)
find_package(ament_cmake_python REQUIRED)
find_package(Boost REQUIRED COMPONENTS system python)
find_package(Eigen3 REQUIRED)
find_package(Python3 REQUIRED COMPONENTS Development NumPy)

include_directories(
  include
  include/numpy_eigen
  ${EIGEN3_INCLUDE_DIRS}
  ${Boost_INCLUDE_DIRS}
  ${Python3_INCLUDE_DIRS}
  ${Python3_NumPy_INCLUDE_DIRS}
)

# Suppress return type warnings in numpy headers
add_compile_options(-fpermissive -D__STRICT_ANSI__)

# Include autogenerated file list
INCLUDE(src/autogen_files.cmake)

# Create Python extension module
add_library(${PROJECT_NAME} SHARED ${AUTOGEN_FILES})

target_include_directories(${PROJECT_NAME} PUBLIC
  $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/include>
  $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/include/numpy_eigen>
  $<INSTALL_INTERFACE:include>
  ${EIGEN3_INCLUDE_DIRS}
  ${Boost_INCLUDE_DIRS}
  ${Python3_INCLUDE_DIRS}
  ${Python3_NumPy_INCLUDE_DIRS}
)

target_link_libraries(${PROJECT_NAME}
  ${Boost_LIBRARIES}
  ${Python3_LIBRARIES}
)

# Python extension modules should not have "lib" prefix, but the actual module is named libnumpy_eigen
set_target_properties(${PROJECT_NAME} PROPERTIES
  PREFIX ""
  OUTPUT_NAME "libnumpy_eigen"
)

# Install Python source files first
ament_python_install_package(${PROJECT_NAME}
  PACKAGE_DIR python/${PROJECT_NAME}
)

# Install the Python module .so to the same location as the __init__.py
install(TARGETS ${PROJECT_NAME}
  LIBRARY DESTINATION local/lib/python${Python3_VERSION_MAJOR}.${Python3_VERSION_MINOR}/dist-packages/numpy_eigen
)

# Install headers
install(DIRECTORY include/
  DESTINATION include
)

ament_export_dependencies(Boost Eigen3 Python3)
ament_export_include_directories(include)

ament_package()
